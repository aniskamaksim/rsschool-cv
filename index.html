function collectAllNumbersInInterval(firstNumber, secondNumber) {
let result = " ";
    if (firstNumber > secondNumber) {
        for (let i = firstNumber; i >= secondNumber; i--) {
            result += firstNumber-- + " ";
        }
    } else if (secondNumber > firstNumber) {
        for (let i = secondNumber; i >= firstNumber; i--) {
            result += secondNumber-- + " ";
        }
    } else {
        result = firstNumber;
    }
    return result;
}
                        
function stringOfPrimes(givenNumber) {
    const sieveOfEratosthenes = [];
    const result = [];
    for (let i = 2; i < givenNumber; i++) {
        if (!sieveOfEratosthenes[i]) {
            result.push(i);
            for (let j = i * 2; j < givenNumber; j += i) {
                sieveOfEratosthenes[j] = true;
            }
        }
    }
    return result.join(" ");
}
